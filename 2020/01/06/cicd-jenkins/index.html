node(‘maven’) {
def mvnCmd = “mvn -s ./nexus_openshift_settings.xml”

stage(‘Checkout Source’) {
git ‘http://gogs-cicd-demo-tools.apps.ocp-test.com/gogs/pt01.git’
sh “ls -l”
}

sh “{mvnCmd} clean package -DskipTests" } stage('Unit Tests') { echo "Unit Tests" sh "{mvnCmd} test”
}

sh “echo JOBBASENAME"sh"{JOB_BASE_NAME}" sh "JOBB​ASEN​AME"sh"{mvnCmd} sonar:sonar -Dsonar.host.url=http://sonarqube.cicd-demo-tools.svc:9000/ -Dsonar.projectName=${JOB_BASE_NAME}”
}

stage(‘Publish to Nexus’) {
echo “Publish to Nexus”
sh “${mvnCmd} deploy -DskipTests=true -DaltDeploymentRepository=nexus::default::http://nexus3-cicd-demo-tools.apps.ocp-test.com/repository/maven-releases/”
}

stage(‘Build OpenShift Image’) {
sh “cp ./target/pipeline-test-1.0.war ./ROOT.war”

}
